#include <stdio.h>
int sum=0,rem;
int reverse_function(int num)
{
  if(num)
  {
    rem=num%10;
    sum=sum*10+rem;
    reverse_function(num/10);
  }
  else
  {
    return sum;
  }
}
int main(){
  int num,rev;
  printf("Enter any number:");
  scanf("%d",&num);
  rev=reverse_function(num);
  printf("The reverse of number is :%d",rev);

}
