#include <stdio.h>
int main()
{
    char str[30];
    char Search;
    int i=0, count=0;
    printf("Enter any string: ");
    gets(str);
    printf("Enter any character to search: ");
    scanf("%c",&Search);
   //Search = getchar();
    while(str[i] != '\0')
    {
        if(str[i] == Search)
        {
            count++;
        }

        i++;
    }
    printf("Total occurrence of '%c' = %d", Search, count);

}
