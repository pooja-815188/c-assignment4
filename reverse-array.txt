
#include <stdio.h>
#define N 5 

int main() 
{
    int arr[N], *ptr;  
    printf("Enter the array: ");
    for (int i = 0; i< N; i++)
    {
        scanf("%d", &arr[i]);
    }
    ptr=&arr[N - 1];
    printf("Reversed array: ");
    for(int i = 0; i < N; i++)
    {
        printf("%d ", *ptr--);
    }
}
