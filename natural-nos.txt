#include<stdio.h>  
  
void display(int);  
  
int main()  
{  
    int n;  
    printf("Enter the number\n");  
    scanf("%d", &n); 
    printf("Natural Numbers from 1 To %d are:",n);  
    display(n);  
  
    return 0;  
}  
  
void display(int num)  
{  
    if(num)  
    {
        display(num-1);  
    }
    else  
    {
        return;  
    }
    printf("\n%d\n", num);  
}
